99 inline void map_widget::timerEvent(QTimerEvent* event)
100 {
101  if (event->timerId() != timer_.timerId())
102  QWidget::timerEvent(event);
103  else if (future_ref_.resultCount() && future_ref_.isResultReadyAt(0)) {
104  decltype(future_ref_) tmp;
105  std::swap(tmp, future_ref_);
106  set_georeference([&] { return tmp.resultAt(0); });
107  start_rendering();
108  }
109  else if (auto render = render_.lock()) {
110  if (auto map = render->get_recent(); !map.img.isNull()) {
111  map_ = std::move(map);
112  update();
113  }
114  }
115  else if (future_map_.valid()) {
116  if (future_ref_.isCanceled())
117  idle_event();
118  map_ = future_map_.get();
119  update();
120  }
121 }

123 inline void map_widget::paintEvent(QPaintEvent*)
124 {
125  if (size() != ref_.size)
126  set_georeference([&] { return ref_ | set_size(size()); });
127  else if (ref_ == map_.ref) {
128  QPainter painter(this);

131  else if (ref_.projection == map_.ref.projection) {
132  QPainter painter(this);

134  painter.save();

138  -1, -1, 1, 1);
