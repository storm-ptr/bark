115  auto load_table(const qualified_name& tbl_nm)
116  {
117  auto bld = builder(*this);
118  bld << "SELECT * FROM " << tbl_nm << " LIMIT 0";
119  auto qry = dataset{file_}.layer_by_sql(bld.sql()).table();
120  auto tbl = dataset{file_}.layer_by_name(tbl_nm).table();
121  auto diff =
122  qry.columns | boost::adaptors::filtered([&](auto& col) {
123  return db::find(tbl.columns, col.name) == std::end(tbl.columns);
124  });
125  auto it = boost::range::find_if(
126  tbl.columns, std::not_fn(same{meta::column_type::Geometry}));
127  auto count = tbl.columns.size();

129  if (tbl.columns.size() - count == 1)
